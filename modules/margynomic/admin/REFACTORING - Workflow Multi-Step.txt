1️⃣ WorkflowEngine.php (Orchestrator)
php/**
 * Pipeline manager - coordina esecuzione step
 */
class WorkflowEngine
{
    private $steps = [];
    private $context = [];
    private $results = [];
    
    /**
     * Esegue workflow completo
     */
    public function execute(array $fields, array $context): array
    {
        // Initialize context
        $this->context = $context;
        
        // Step 1: Keyword Research (Gemini 3 Pro)
        $keywords = $this->executeStep(new Step1_KeywordResearch());
        $this->context['keywords'] = $keywords;
        
        // Step 2-5: Content Generation (Haiku)
        foreach ($fields as $fieldName) {
            $step = $this->getStepForField($fieldName);
            $this->results[$fieldName] = $this->executeStep($step);
        }
        
        return $this->results;
    }
    
    /**
     * Esegue singolo step con error handling
     */
    private function executeStep(WorkflowStep $step): mixed
    {
        try {
            return $step->execute($this->context);
        } catch (Exception $e) {
            CentralLogger::error('workflow', 'Step failed', [
                'step' => get_class($step),
                'error' => $e->getMessage()
            ]);
            
            // Fallback strategy
            return $step->fallback($this->context);
        }
    }
}

2️⃣ WorkflowStep.php (Abstract Base)
php/**
 * Base class per tutti gli step workflow
 */
abstract class WorkflowStep
{
    protected $llmClient;
    protected $config;
    
    /**
     * Esegue step
     */
    abstract public function execute(array $context): mixed;
    
    /**
     * Strategia fallback se step fallisce
     */
    abstract public function fallback(array $context): mixed;
    
    /**
     * Valida output step
     */
    abstract protected function validate($output): bool;
    
    /**
     * Retry logic con backoff
     */
    protected function executeWithRetry(callable $fn, int $maxAttempts = 3): mixed
    {
        for ($attempt = 1; $attempt <= $maxAttempts; $attempt++) {
            try {
                $result = $fn();
                
                if ($this->validate($result)) {
                    return $result;
                }
                
                CentralLogger::warning('workflow_step', 'Validation failed', [
                    'attempt' => $attempt
                ]);
                
            } catch (Exception $e) {
                if ($attempt === $maxAttempts) {
                    throw $e;
                }
                
                sleep(pow(2, $attempt)); // Exponential backoff
            }
        }
        
        throw new Exception('Max retry attempts exceeded');
    }
}
Responsabilità:

✅ Template method pattern
✅ Retry logic standardizzato
✅ Validation framework

3️⃣ Step1_KeywordResearch.php (Gemini 3 Pro)
php/**
 * STEP 1: Competitor Analysis + Keyword Extraction
 * Model: Gemini 3 Pro Preview (extended thinking)
 */
class Step1_KeywordResearch extends WorkflowStep
{
    public function execute(array $context): array
    {
        return $this->executeWithRetry(function() use ($context) {
            
            // Load prompt template
            $prompt = $this->buildPrompt($context);
            
            // Call Gemini 3 Pro con extended thinking
            $response = $this->llmClient->generate($prompt, [
                'model' => 'gemini-3-pro-preview',
                'temperature' => 0.3,
                'max_tokens' => 3000,
                'thinking' => true  // Enable extended thinking
            ]);
            
            // Parse JSON response
            $parser = new KeywordParser();
            $keywords = $parser->parse($response);
            
            // Post-processing: dedup, sort, filter
            return $this->processKeywords($keywords, $context);
        });
    }
    
    private function buildPrompt(array $context): string
    {
        // Load template da file
        $template = file_get_contents(__DIR__ . '/../prompts/keyword_research.txt');
        
        // Replace placeholders
        return strtr($template, [
            '{COMPETITOR_DATA}' => $this->formatCompetitors($context),
            '{PRODUCT_TYPE}' => $context['product_type'] ?? 'generic',
            '{CATEGORY}' => $context['category'] ?? 'grocery'
        ]);
    }
    
    private function processKeywords(array $keywords, array $context): array
    {
        // Deduplication
        $keywords = array_unique($keywords);
        
        // Scoring (frequency × position weight)
        $scored = $this->scoreKeywords($keywords, $context);
        
        // Sort by score DESC
        usort($scored, fn($a, $b) => $b['score'] <=> $a['score']);
        
        // Return top 100
        return array_slice(array_column($scored, 'keyword'), 0, 100);
    }
    
    public function fallback(array $context): array
    {
        // Fallback: estrai da titolo/descrizione esistenti
        return $this->extractFromExistingContent($context);
    }
    
    protected function validate($output): bool
    {
        return is_array($output) && count($output) >= 50;
    }
}

php/**
 * STEP 2: Title Generation
 * Model: Haiku 4.5 (fast + economico)
 */
class Step2_TitleGeneration extends WorkflowStep
{
    public function execute(array $context): array
    {
        return $this->executeWithRetry(function() use ($context) {
            
            $prompt = $this->buildPrompt($context);
            
            // Call Haiku (veloce ed economico)
            $response = $this->llmClient->generate($prompt, [
                'model' => 'haiku-4.5',
                'temperature' => 0.7,
                'max_tokens' => 100,
                'stop' => ["\n", "\n\n"]
            ]);
            
            // Parse e cleanup
            $title = $this->parseTitle($response);
            
            // Template enforcement
            $title = $this->enforceTemplate($title, $context);
            
            // Validation
            $validator = new ContentValidator();
            $validation = $validator->validate('item_name', $title);
            
            return [
                'content' => $title,
                'success' => $validation['valid'],
                'validation' => $validation
            ];
        });
    }
    
    private function buildPrompt(array $context): string
    {
        $template = file_get_contents(__DIR__ . '/../prompts/title_generation.txt');
        
        // Keywords top 20
        $keywords = array_slice($context['keywords'] ?? [], 0, 20);
        
        return strtr($template, [
            '{BRAND}' => $context['brand'],
            '{WEIGHT}' => $context['weight'],
            '{KEYWORDS}' => implode(', ', $keywords),
            '{TEMPLATE}' => $this->getTemplateForCategory($context['category'])
        ]);
    }
    
    private function enforceTemplate(string $title, array $context): string
    {
        $brand = $context['brand'];
        $weight = $context['weight'];
        
        // Parse componenti
        $parsed = $this->parseComponents($title);
        
        // Reassemble con template FISSO
        return sprintf(
            "%s - %s %s | %s | %s",
            $brand,
            $parsed['product'],
            $weight,
            $parsed['feature'],
            $parsed['benefit']
        );
    }
    
    public function fallback(array $context): array
    {
        // Fallback: template generico
        return [
            'content' => sprintf(
                "%s - %s %s | Premium | Per Uso Quotidiano",
                $context['brand'],
                $context['product_type'] ?? 'Prodotto',
                $context['weight']
            ),
            'success' => false,
            'validation' => ['errors' => ['Fallback generico']]
        ];
    }
    
    protected function validate($output): bool
    {
        return isset($output['content']) && 
               strlen($output['content']) >= 150 &&
               strpos($output['content'], $this->context['brand']) === 0;
    }
}

6️⃣ Step4_BulletsGeneration.php (Haiku)
php/**
 * STEP 4: Bullets Generation (5 bullets sequenziali)
 * Model: Haiku 4.5 (fast batch generation)
 */
class Step4_BulletsGeneration extends WorkflowStep
{
    private $focuses = [
        'Qualità e Origine',
        'Processo Produzione',
        'Versatilità Uso',
        'Benefici Nutrizionali',
        'Garanzia e Certificazioni'
    ];
    
    public function execute(array $context): array
    {
        $bullets = [];
        $usedKeywords = $this->extractUsedKeywords($context);
        
        foreach ($this->focuses as $i => $focus) {
            $bullet = $this->generateSingleBullet(
                $i + 1,
                $focus,
                $context,
                $usedKeywords
            );
            
            $bullets["bullet_point" . ($i + 1)] = $bullet;
            $usedKeywords = array_merge($usedKeywords, $this->extractKeywords($bullet['content']));
        }
        
        return $bullets;
    }
    
    private function generateSingleBullet(int $index, string $focus, array $context, array $usedKeywords): array
    {
        $template = file_get_contents(__DIR__ . '/../prompts/bullet_generation.txt');
        
        $availableKeywords = array_diff($context['keywords'], $usedKeywords);
        
        $prompt = strtr($template, [
            '{BULLET_NUMBER}' => $index,
            '{FOCUS}' => $focus,
            '{KEYWORDS}' => implode(', ', array_slice($availableKeywords, 0, 15)),
            '{USED_KEYWORDS}' => implode(', ', $usedKeywords)
        ]);
        
        $response = $this->llmClient->generate($prompt, [
            'model' => 'haiku-4.5',
            'temperature' => 0.7,
            'max_tokens' => 150,
            'stop' => ["\n\n"]
        ]);
        
        $bullet = $this->enforceBulletFormat($response);
        
        $validator = new ContentValidator();
        $validation = $validator->validate('bullet_point1', $bullet);
        
        return [
            'content' => $bullet,
            'success' => $validation['valid'],
            'validation' => $validation
        ];
    }
    
    public function fallback(array $context): array
    {
        // Genera bullets generici
        $genericBullets = [
            "✓ QUALITÀ PREMIUM: Prodotto selezionato con cura per garantire standard elevati.",
            "✓ VERSATILE: Ideale per molteplici utilizzi in cucina e non solo.",
            "✓ NATURALE: Ingredienti scelti accuratamente senza additivi artificiali.",
            "✓ FACILE DA USARE: Pronto all'uso per massima comodità.",
            "✓ SODDISFAZIONE GARANTITA: Prodotto testato e certificato."
        ];
        
        $result = [];
        foreach ($genericBullets as $i => $bullet) {
            $result["bullet_point" . ($i + 1)] = [
                'content' => $bullet,
                'success' => false,
                'validation' => ['warnings' => ['Fallback generico']]
            ];
        }
        
        return $result;
    }
    
    protected function validate($output): bool
    {
        return count($output) === 5 && 
               array_reduce($output, fn($carry, $item) => $carry && $item['success'], true);
    }
}

php/**
 * STEP 5: Hidden Keywords
 * NO LLM - Pure code logic
 */
class Step5_HiddenKeywords extends WorkflowStep
{
    public function execute(array $context): array
    {
        // Raccogli tutte le keywords usate
        $usedKeywords = [];
        
        // Da titolo
        $usedKeywords = array_merge(
            $usedKeywords,
            $this->extractKeywords($context['generated_title'])
        );
        
        // Da description
        $usedKeywords = array_merge(
            $usedKeywords,
            $this->extractKeywords($context['generated_description'])
        );
        
        // Da bullets
        foreach ($context['generated_bullets'] as $bullet) {
            $usedKeywords = array_merge(
                $usedKeywords,
                $this->extractKeywords($bullet['content'])
            );
        }
        
        // Rimuovi duplicati
        $usedKeywords = array_unique(array_map('strtolower', $usedKeywords));
        
        // Filtra keywords disponibili
        $availableKeywords = array_diff(
            array_map('strtolower', $context['keywords']),
            $usedKeywords
        );
        
        // Top 30 keywords non usate
        $hiddenKeywords = array_slice($availableKeywords, 0, 30);
        
        // Formato: spazi, no virgole
        $formatted = implode(' ', $hiddenKeywords);
        
        // Enforce lunghezza 150-180
        if (strlen($formatted) < 150) {
            // Aggiungi sinonimi se troppo corto
            $formatted = $this->expandWithSynonyms($formatted, 150);
        }
        
        if (strlen($formatted) > 180) {
            // Trunca a parola intera
            $formatted = substr($formatted, 0, 177);
            $lastSpace = strrpos($formatted, ' ');
            $formatted = substr($formatted, 0, $lastSpace);
        }
        
        return [
            'content' => $formatted,
            'success' => true,
            'validation' => ['valid' => true]
        ];
    }
    
    private function extractKeywords(string $text): array
    {
        // Remove HTML
        $text = strip_tags($text);
        
        // Tokenize
        $words = preg_split('/\s+/', strtolower($text));
        
        // Filter stopwords e parole corte
        $stopwords = ['il', 'la', 'di', 'da', 'per', 'con', 'su', 'a', 'e'];
        $words = array_filter($words, fn($w) => 
            strlen($w) > 3 && !in_array($w, $stopwords)
        );
        
        return array_values($words);
    }
    
    public function fallback(array $context): array
    {
        return [
            'content' => 'qualità premium naturale artigianale tradizionale genuino italiano',
            'success' => false,
            'validation' => ['warnings' => ['Fallback keywords generiche']]
        ];
    }
    
    protected function validate($output): bool
    {
        $length = strlen($output['content']);
        return $length >= 150 && $length <= 180;
    }
}
